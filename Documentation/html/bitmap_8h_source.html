113  class CColor {
114 public:
115  static inline unsigned int BitCountByMask(unsigned int Mask) {
116  unsigned int BitCount = 0;
117  while (Mask) {
118  Mask &= Mask - 1;
119  BitCount++;
120  }
121  return BitCount;
122  }
123 
124  static inline unsigned int BitPositionByMask(unsigned int Mask) {
125  return BitCountByMask((Mask & (~Mask + 1)) - 1);
126  }
127 
128  static inline unsigned int ComponentByMask(unsigned int Color, unsigned int Mask) {
129  unsigned int Component = Color & Mask;
130  return Component >> BitPositionByMask(Mask);
131  }
132 
133  static inline unsigned int BitCountToMask(unsigned int BitCount) {
134  return (BitCount == 32) ? 0xFFFFFFFF : (1 << BitCount) - 1;
135  }
136 
137  static unsigned int Convert(unsigned int Color, unsigned int FromBitCount, unsigned int ToBitCount) {
138  if (ToBitCount < FromBitCount) {
139  Color >>= (FromBitCount - ToBitCount);
140  } else {
141  Color <<= (ToBitCount - FromBitCount);
142  if (Color > 0) {
143  Color |= BitCountToMask(ToBitCount - FromBitCount);
144  }
145  }
146  return Color;
147  }

175  bool Load(const char *Filename) {
176  std::ifstream file(Filename, std::ios::binary | std::ios::in);
177 
178  if (file.bad()) {
179  return false;
180  }
181 
182  if (file.is_open() == false) {
183  return false;
184  }
185 
186  Dispose();
187 
188  file.read((char*) &m_BitmapFileHeader, BITMAP_FILEHEADER_SIZE);
189  if (m_BitmapFileHeader.Signature != BITMAP_SIGNATURE) {
190  return false;
191  }
192 
193  file.read((char*) &m_BitmapHeader, sizeof(BITMAP_HEADER));
194 
195  /* Load Color Table */
196 
197  file.seekg(BITMAP_FILEHEADER_SIZE + m_BitmapHeader.HeaderSize, std::ios::beg);
198 
199  unsigned int ColorTableSize = 0;
200 
201  if (m_BitmapHeader.BitCount == 1) {
202  ColorTableSize = 2;
203  } else if (m_BitmapHeader.BitCount == 4) {
204  ColorTableSize = 16;
205  } else if (m_BitmapHeader.BitCount == 8) {
206  ColorTableSize = 256;
207  }
208 
209  // Always allocate full sized color table
210 
211  BGRA* ColorTable = new BGRA[ColorTableSize]; // std::bad_alloc exception should be thrown if memory is not available
212 
213  file.read((char*) ColorTable, sizeof(BGRA) * m_BitmapHeader.ClrUsed);
214 
215  /* ... Color Table for 16 bits images are not supported yet */
216 
217  m_BitmapSize = GetWidth() * GetHeight();
218  m_BitmapData = new RGBA[m_BitmapSize];
219 
220  unsigned int LineWidth = ((GetWidth() * GetBitCount() / 8) + 3) & ~3;
221  uint8_t *Line = new uint8_t[LineWidth];
222 
223  file.seekg(m_BitmapFileHeader.BitsOffset, std::ios::beg);
224 
225  int Index = 0;
226  bool Result = true;
227 
228  if (m_BitmapHeader.Compression == 0) {
229  for (unsigned int i = 0; i < GetHeight(); i++) {
230  file.read((char*) Line, LineWidth);
231 
232  uint8_t *LinePtr = Line;
233 
234  for (unsigned int j = 0; j < GetWidth(); j++) {
235  if (m_BitmapHeader.BitCount == 1) {
236  uint32_t Color = *((uint8_t*) LinePtr);
237  for (int k = 0; k < 8; k++) {
238  m_BitmapData[Index].Red = ColorTable[Color & 0x80 ? 1 : 0].Red;
239  m_BitmapData[Index].Green = ColorTable[Color & 0x80 ? 1 : 0].Green;
240  m_BitmapData[Index].Blue = ColorTable[Color & 0x80 ? 1 : 0].Blue;
241  m_BitmapData[Index].Alpha = ColorTable[Color & 0x80 ? 1 : 0].Alpha;
242  Index++;
243  Color <<= 1;
244  }
245  LinePtr++;
246  j += 7;
247  } else if (m_BitmapHeader.BitCount == 4) {
248  uint32_t Color = *((uint8_t*) LinePtr);
249  m_BitmapData[Index].Red = ColorTable[(Color >> 4) & 0x0f].Red;
250  m_BitmapData[Index].Green = ColorTable[(Color >> 4) & 0x0f].Green;
251  m_BitmapData[Index].Blue = ColorTable[(Color >> 4) & 0x0f].Blue;
252  m_BitmapData[Index].Alpha = ColorTable[(Color >> 4) & 0x0f].Alpha;
253  Index++;
254  m_BitmapData[Index].Red = ColorTable[Color & 0x0f].Red;
255  m_BitmapData[Index].Green = ColorTable[Color & 0x0f].Green;
256  m_BitmapData[Index].Blue = ColorTable[Color & 0x0f].Blue;
257  m_BitmapData[Index].Alpha = ColorTable[Color & 0x0f].Alpha;
258  Index++;
259  LinePtr++;
260  j++;
261  } else if (m_BitmapHeader.BitCount == 8) {
262  uint32_t Color = *((uint8_t*) LinePtr);
263  m_BitmapData[Index].Red = ColorTable[Color].Red;
264  m_BitmapData[Index].Green = ColorTable[Color].Green;
265  m_BitmapData[Index].Blue = ColorTable[Color].Blue;
266  m_BitmapData[Index].Alpha = ColorTable[Color].Alpha;
267  Index++;
268  LinePtr++;
269  } else if (m_BitmapHeader.BitCount == 16) {
270  uint32_t Color = *((uint16_t*) LinePtr);
271  m_BitmapData[Index].Red = ((Color >> 10) & 0x1f) << 3;
272  m_BitmapData[Index].Green = ((Color >> 5) & 0x1f) << 3;
273  m_BitmapData[Index].Blue = (Color & 0x1f) << 3;
274  m_BitmapData[Index].Alpha = 255;
275  Index++;
276  LinePtr += 2;
277  } else if (m_BitmapHeader.BitCount == 24) {
278  uint32_t Color = *((uint32_t*) LinePtr);
279  m_BitmapData[Index].Blue = Color & 0xff;
280  m_BitmapData[Index].Green = (Color >> 8) & 0xff;
281  m_BitmapData[Index].Red = (Color >> 16) & 0xff;
282  m_BitmapData[Index].Alpha = 255;
283  Index++;
284  LinePtr += 3;
285  } else if (m_BitmapHeader.BitCount == 32) {
286  uint32_t Color = *((uint32_t*) LinePtr);
287  m_BitmapData[Index].Blue = Color & 0xff;
288  m_BitmapData[Index].Green = (Color >> 8) & 0xff;
289  m_BitmapData[Index].Red = (Color >> 16) & 0xff;
290  m_BitmapData[Index].Alpha = Color >> 24;
291  Index++;
292  LinePtr += 4;
293  }
294  }
295  }
296  } else if (m_BitmapHeader.Compression == 1) { // RLE 8
297  uint8_t Count = 0;
298  uint8_t ColorIndex = 0;
299  int x = 0, y = 0;
300 
301  while (file.eof() == false) {
302  file.read((char*) &Count, sizeof(uint8_t));
303  file.read((char*) &ColorIndex, sizeof(uint8_t));
304 
305  if (Count > 0) {
306  Index = x + y * GetWidth();
307  for (int k = 0; k < Count; k++) {
308  m_BitmapData[Index + k].Red = ColorTable[ColorIndex].Red;
309  m_BitmapData[Index + k].Green = ColorTable[ColorIndex].Green;
310  m_BitmapData[Index + k].Blue = ColorTable[ColorIndex].Blue;
311  m_BitmapData[Index + k].Alpha = ColorTable[ColorIndex].Alpha;
312  }
313  x += Count;
314  } else if (Count == 0) {
315  int Flag = ColorIndex;
316  if (Flag == 0) {
317  x = 0;
318  y++;
319  } else if (Flag == 1) {
320  break;
321  } else if (Flag == 2) {
322  char rx = 0;
323  char ry = 0;
324  file.read((char*) &rx, sizeof(char));
325  file.read((char*) &ry, sizeof(char));
326  x += rx;
327  y += ry;
328  } else {
329  Count = Flag;
330  Index = x + y * GetWidth();
331  for (int k = 0; k < Count; k++) {
332  file.read((char*) &ColorIndex, sizeof(uint8_t));
333  m_BitmapData[Index + k].Red = ColorTable[ColorIndex].Red;
334  m_BitmapData[Index + k].Green = ColorTable[ColorIndex].Green;
335  m_BitmapData[Index + k].Blue = ColorTable[ColorIndex].Blue;
336  m_BitmapData[Index + k].Alpha = ColorTable[ColorIndex].Alpha;
337  }
338  x += Count;
339  // Attention: Current Microsoft STL implementation seems to be buggy, tellg() always returns 0.
340  if (file.tellg() & 1) {
341  file.seekg(1, std::ios::cur);
342  }
343  }
344  }
345  }
346  } else if (m_BitmapHeader.Compression == 2) { // RLE 4
347  /* RLE 4 is not supported */
348  Result = false;
349  } else if (m_BitmapHeader.Compression == 3) { // BITFIELDS
350 
351  /* We assumes that mask of each color component can be in any order */
352 
353  uint32_t BitCountRed = CColor::BitCountByMask(m_BitmapHeader.RedMask);
354  uint32_t BitCountGreen = CColor::BitCountByMask(m_BitmapHeader.GreenMask);
355  uint32_t BitCountBlue = CColor::BitCountByMask(m_BitmapHeader.BlueMask);
356  uint32_t BitCountAlpha = CColor::BitCountByMask(m_BitmapHeader.AlphaMask);
357 
358  for (unsigned int i = 0; i < GetHeight(); i++) {
359  file.read((char*) Line, LineWidth);
360 
361  uint8_t *LinePtr = Line;
362 
363  for (unsigned int j = 0; j < GetWidth(); j++) {
364 
365  uint32_t Color = 0;
366 
367  if (m_BitmapHeader.BitCount == 16) {
368  Color = *((uint16_t*) LinePtr);
369  LinePtr += 2;
370  } else if (m_BitmapHeader.BitCount == 32) {
371  Color = *((uint32_t*) LinePtr);
372  LinePtr += 4;
373  } else {
374  // Other formats are not valid
375  }
376  m_BitmapData[Index].Red = CColor::Convert(CColor::ComponentByMask(Color, m_BitmapHeader.RedMask), BitCountRed, 8);
377  m_BitmapData[Index].Green = CColor::Convert(CColor::ComponentByMask(Color, m_BitmapHeader.GreenMask), BitCountGreen, 8);
378  m_BitmapData[Index].Blue = CColor::Convert(CColor::ComponentByMask(Color, m_BitmapHeader.BlueMask), BitCountBlue, 8);
379  m_BitmapData[Index].Alpha = CColor::Convert(CColor::ComponentByMask(Color, m_BitmapHeader.AlphaMask), BitCountAlpha, 8);
380 
381  Index++;
382  }
383  }
384  }
385 
386  delete [] ColorTable;
387  delete [] Line;
388 
389  file.close();
390  return Result;
391  }

540  /* Copies internal RGBA buffer to user specified buffer and converts it into destination
541  * bit format specified by component masks.
542  *
543  * Typical Bitmap color formats (BGR/BGRA):
544  *
545  * Masks for 16 bit (5-5-5): ALPHA = 0x00000000, RED = 0x00007C00, GREEN = 0x000003E0, BLUE = 0x0000001F
546  * Masks for 16 bit (5-6-5): ALPHA = 0x00000000, RED = 0x0000F800, GREEN = 0x000007E0, BLUE = 0x0000001F
547  * Masks for 24 bit: ALPHA = 0x00000000, RED = 0x00FF0000, GREEN = 0x0000FF00, BLUE = 0x000000FF
548  * Masks for 32 bit: ALPHA = 0xFF000000, RED = 0x00FF0000, GREEN = 0x0000FF00, BLUE = 0x000000FF
549  *
550  * Other color formats (RGB/RGBA):
551  *
552  * Masks for 32 bit (RGBA): ALPHA = 0xFF000000, RED = 0x000000FF, GREEN = 0x0000FF00, BLUE = 0x00FF0000
553  *
554  * Bit count will be rounded to next 8 bit boundary. If IncludePadding is true, it will be ensured
555  * that line width is a multiple of 4. padding bytes are included if necessary.
556  *
557  * NOTE: systems with big endian byte order may require masks in inversion order.
558  */
559 
560  bool GetBits(void* Buffer, unsigned int &Size, unsigned int RedMask, unsigned int GreenMask, unsigned int BlueMask, unsigned int AlphaMask, bool IncludePadding = true) {
561  bool Result = false;
562 
563  uint32_t BitCountRed = CColor::BitCountByMask(RedMask);
564  uint32_t BitCountGreen = CColor::BitCountByMask(GreenMask);
565  uint32_t BitCountBlue = CColor::BitCountByMask(BlueMask);
566  uint32_t BitCountAlpha = CColor::BitCountByMask(AlphaMask);
567 
568  unsigned int BitCount = (BitCountRed + BitCountGreen + BitCountBlue + BitCountAlpha + 7) & ~7;
569 
570  if (BitCount > 32) {
571  return false;
572  }
573 
574  unsigned int w = GetWidth();
575  //unsigned int LineWidth = (w + 3) & ~3;
576  unsigned int dataBytesPerLine = (w * BitCount + 7) / 8;
577  unsigned int LineWidth = (dataBytesPerLine + 3) & ~3;
578 
579  if (Size == 0 || Buffer == 0) {
580  //Size = (LineWidth * GetHeight() * BitCount) / 8 + sizeof(unsigned int);
581  Size = (GetWidth() * GetHeight() * BitCount) / 8 + sizeof(unsigned int);
582  return true;
583  }
584 
585  uint8_t* BufferPtr = (uint8_t*) Buffer;
586 
587  Result = true;
588 
589  uint32_t BitPosRed = CColor::BitPositionByMask(RedMask);
590  uint32_t BitPosGreen = CColor::BitPositionByMask(GreenMask);
591  uint32_t BitPosBlue = CColor::BitPositionByMask(BlueMask);
592  uint32_t BitPosAlpha = CColor::BitPositionByMask(AlphaMask);
593 
594  unsigned int j = 0;
595 
596  for (unsigned int i = 0; i < m_BitmapSize; i++) {
597  *(uint32_t*) BufferPtr =
598  (CColor::Convert(m_BitmapData[i].Blue, 8, BitCountBlue) << BitPosBlue) |
599  (CColor::Convert(m_BitmapData[i].Green, 8, BitCountGreen) << BitPosGreen) |
600  (CColor::Convert(m_BitmapData[i].Red, 8, BitCountRed) << BitPosRed) |
601  (CColor::Convert(m_BitmapData[i].Alpha, 8, BitCountAlpha) << BitPosAlpha);
602 
603  if (IncludePadding) {
604  j++;
605  if (j >= w) {
606  for (unsigned int k = 0; k < LineWidth - dataBytesPerLine; k++) {
607  BufferPtr += (BitCount >> 3);
608  }
609  j = 0;
610  }
611  }
612 
613  BufferPtr += (BitCount >> 3);
614  }
615 
616  Size -= sizeof(unsigned int);
617 
618  return Result;
619  }

629  bool GetBitsWithPalette(void* Buffer, unsigned int &Size, unsigned int BitCount, BGRA* &Palette, unsigned int &PaletteSize, bool OptimalPalette = false, bool IncludePadding = true) {
630  bool Result = false;
631 
632  if (BitCount > 16) {
633  return false;
634  }
